# background visualiser thread reads via getBuffer(). The worst case is a
# single frame of stale/torn visual data, which is imperceptible.
race:osci::BufferConsumer::write
race:osci::BufferConsumer::getBuffer

# OpenGL driver internals

# JUCE OpenGL render thread vs MessageManager::Lock
